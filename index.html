Q.1 Count Characters

        const countCharacters = (S) => 
{
  let countA=0; let countD=0;
  let N=S.length; let B=[];
  for(let i=0; i < N; i++)
  {
    if(S[i]==='A')
    countA++;
    if(S[i]==='D')
    countD++;
  }
  B[0]=countA;
  B[1]=countD;
  return B;
};

Q . 8 Plaindrome Check

 var Palin_Check = (str) =>
{
	let revstr="";
	let N=str.length;
	for(let i=N-1;i>=0;i--)
	{
	    revstr+=str[i];
	}
	if(str===revstr)
	{
	    return 'True';
	}
	else
	{
	    return 'False';
	}
}
